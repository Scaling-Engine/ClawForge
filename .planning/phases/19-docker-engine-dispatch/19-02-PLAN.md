3. Add `waitAndNotify` function in `lib/ai/tools.js` (private, not exported):
   ```javascript
   async function waitAndNotify(container, jobId, threadId) {
     try {
       const { StatusCode } = await waitForContainer(container);
       const logs = await collectLogs(container);

       // Build notification payload matching Actions webhook shape
       const results = {
         job: '', // Could fetch from GitHub but not critical
         pr_url: '',  // Will be populated after PR creation by entrypoint
         run_url: '',
         status: StatusCode === 0 ? 'completed' : 'failure',
         failure_stage: StatusCode !== 0 ? 'container_execution' : '',
         merge_result: '',
         log: logs?.stdout?.slice(-4000) || '',
         changed_files: [],
         commit_message: '',
         target_repo: '',
       };

       // Query GitHub for PR info (entrypoint creates PR before exiting)
       try {
         const { fetchRepoFile, githubApi } = await import('../tools/github.js');
         const { GH_OWNER, GH_REPO } = process.env;
         const branch = `job/${jobId}`;
         const prs = await githubApi(`/repos/${GH_OWNER}/${GH_REPO}/pulls?head=${GH_OWNER}:${branch}&state=all`);
         if (prs.length > 0) {
           results.pr_url = prs[0].html_url;
           results.merge_result = prs[0].merged_at ? 'merged' : 'open';
         }
         // Fetch job.md content
         try {
           results.job = await fetchRepoFile(GH_OWNER, GH_REPO, `logs/${jobId}/job.md`, { ref: branch });
         } catch {}
       } catch {}

       // Reuse existing notification pipeline
       const message = await summarizeJob(results);
       await createNotification(message, { job_id: jobId, ...results });

       // Mark as notified in DB (for dedup against Actions webhook)
       markDockerJobNotified(jobId);

       // Route to originating thread (same logic as handleGithubWebhook)
       if (threadId) {
         const origin = getJobOrigin(jobId);
         if (origin) {
           saveJobOutcome({
             jobId,
             threadId: origin.threadId,
             status: results.status,
             mergeResult: results.merge_result,
             prUrl: results.pr_url,
             targetRepo: results.target_repo || null,
             changedFiles: results.changed_files,
             logSummary: message,
           });

           const { addToThread } = await import('../ai/index.js');
           addToThread(origin.threadId, `[Job completed] ${message}`).catch(() => {});

           if (origin.platform === 'slack') {
             const [channel, threadTs] = origin.threadId.split(':');
             const { SLACK_BOT_TOKEN } = process.env;
             if (SLACK_BOT_TOKEN && channel && threadTs) {
               try {
                 const { WebClient } = await import('@slack/web-api');
                 const slack = new WebClient(SLACK_BOT_TOKEN);
                 await slack.chat.postMessage({ channel, thread_ts: threadTs, text: message });
               } catch (err) { console.error('Docker job Slack notification failed:', err); }
             }
           }
           if (origin.platform === 'telegram') {
             const { TELEGRAM_BOT_TOKEN } = process.env;
             if (TELEGRAM_BOT_TOKEN && origin.threadId) {
               try {
                 const { sendMessage } = await import('../lib/tools/telegram.js');
                 await sendMessage(TELEGRAM_BOT_TOKEN, origin.threadId, message);
               } catch (err) { console.error('Docker job Telegram notification failed:', err); }
             }
           }
         }
       }

       // Cleanup container
       await removeContainer(container);
       console.log(`Docker job ${jobId.slice(0, 8)} completed (exit ${StatusCode}), notified, cleaned up`);
     } catch (err) {
       console.error(`Docker job ${jobId.slice(0, 8)} wait/notify failed:`, err);
       try { await removeContainer(container); } catch {}
     }
   }
   ```

   IMPORTANT: The Telegram import path inside waitAndNotify should be `'../../lib/tools/telegram.js'` if tools.js is in `lib/ai/`, or use the correct relative path. Check the existing import in api/index.js for reference -- it uses `'../lib/tools/telegram.js'` because api/ is at root level. From `lib/ai/tools.js`, the correct path is `'../tools/telegram.js'`.
