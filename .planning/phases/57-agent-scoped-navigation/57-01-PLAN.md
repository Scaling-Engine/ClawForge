Create `templates/app/agent/[slug]/layout.js` as a Next.js async Server Component layout.

This file is 2 dirs deep from `templates/app/`, so lib imports use `../../lib/`.

The layout must:
1. Call `auth()` from `../../lib/auth/index.js` to get the session
2. Await `params` to extract `slug` (Next.js 15 pattern — params is a Promise)
3. Check if the user is assigned to this agent:
   - If `session.user.role === 'admin'` OR `SUPERADMIN_HUB !== 'true'`, allow access (admins and spoke-mode instances bypass assignment check)
   - If `session.user.assignedAgents` is an array and it includes `slug`, allow access
   - Otherwise: redirect to `/agents` (use `redirect` from `next/navigation`)
4. Import `PageLayout` from `../../lib/chat/components/index.js`
5. Pass `session` and `agentSlug={slug}` to `PageLayout` — the PageLayout component does NOT yet accept agentSlug, so pass it as a data attribute or just pass session for now. Actually: render a thin wrapper that passes children inside the existing PageLayout pattern. Use the EXACT same pattern as `page-layout.js` — import AppSidebar, SidebarProvider, SidebarInset, ChatNavProvider from the package.
6. The `navigateToChat` function should navigate to `/agent/${slug}/chat/${id}` (agent-scoped chat URL)
7. Export the layout as default

IMPORTANT: This is a Next.js layout (not a page) — it must export a default function that accepts `{ children, params }`. The params object must be awaited.

```js
import { auth } from '../../lib/auth/index.js';
import { redirect } from 'next/navigation';
import { AppSidebar } from '../../lib/chat/components/index.js';
import { SidebarProvider, SidebarInset } from '../../lib/chat/components/index.js';
import { ChatNavProvider } from '../../lib/chat/components/index.js';

export default async function AgentLayout({ children, params }) {
  const { slug } = await params;
  const session = await auth();

  // Validate agent access
  const isAdmin = session?.user?.role === 'admin';
  const isHubMode = process.env.SUPERADMIN_HUB === 'true';
  const assignedAgents = session?.user?.assignedAgents ?? [];
  const hasAccess = !isHubMode || isAdmin || assignedAgents.includes(slug);

  if (!hasAccess) {
    redirect('/agents');
  }

  function navigateToChat(id) {
    // Client-side navigation — cannot use this in server component directly.
    // PageLayout will override navigateToChat; define the default here as a string template.
    if (typeof window !== 'undefined') {
      window.location.href = id ? `/agent/${slug}/chat/${id}` : `/agent/${slug}/chat`;
    }
  }

  return (
    <ChatNavProvider value={{ activeChatId: null, navigateToChat: (id) => {
      if (typeof window !== 'undefined') {
        window.location.href = id ? `/agent/${slug}/chat/${id}` : `/agent/${slug}/chat`;
      }
    }}}>
      <SidebarProvider>
        <AppSidebar user={session?.user} agentSlug={slug} />
        <SidebarInset>
          <div className="flex flex-col h-full max-w-4xl mx-auto w-full px-4 py-6">
            {children}
          </div>
        </SidebarInset>
      </SidebarProvider>
    </ChatNavProvider>
  );
}
```

- Make a thin `AgentChatNavProvider` client component, OR
- Use the existing `PageLayout` component but pass it `agentSlug` so it can build the correct nav URL.

2. The server layout calls auth, validates access, then renders `<AgentLayoutClient agentSlug={slug} user={session.user}>{children}</AgentLayoutClient>`.
