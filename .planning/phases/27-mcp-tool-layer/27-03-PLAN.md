2. Add new exported async function `getMcpServers()`:
   ```javascript
   export async function getMcpServers() {
     await requireAuth();
     const servers = loadMcpServers();
     // Redact env values — never expose credentials in UI (MCP-09)
     return servers.map(({ name, command, args, allowedTools, hydrateTools }) => ({
       name,
       command,
       args: args || [],
       allowedTools: allowedTools || [],
       hydrateTools: hydrateTools || [],
       // env deliberately omitted — credentials never sent to client
     }));
   }
   ```
