       const logger = pino({

           ? { target: 'pino-pretty', options: { colorize: true } }

       });

       }

         Allowlist keys: `['route', 'jobId', 'threadId', 'platform', 'statusCode', 'code']`
         Return object with only those keys that exist in meta. Never store message text or API keys.
