**getAllAgentPullRequests(state = 'open'):**
```js
export async function getAllAgentPullRequests(state = 'open') {
  const user = await requireAuth();
  const { queryAllInstances } = await import('../superadmin/client.js');
  const results = await queryAllInstances('prs', { state });
  const isHubAdmin = process.env.SUPERADMIN_HUB === 'true' && user.role === 'admin';
  const assignedSlugs = new Set(user.assignedAgents ?? []);

  const rows = [];
  for (const r of results) {
    if (!isHubAdmin && !assignedSlugs.has(r.instance)) continue;
    if (r.error || !r.data?.prs) {
      rows.push({ agentSlug: r.instance, stale: true, error: r.error ?? 'offline', prs: [] });
    } else {
      rows.push({ agentSlug: r.instance, stale: false, error: null, prs: r.data.prs });
    }
  }
  return rows;
}
```

**dispatchAgentJob(agentSlug, jobDescription):**
```js
export async function dispatchAgentJob(agentSlug, jobDescription) {
  const user = await requireAuth();
  if (!agentSlug || typeof agentSlug !== 'string') return { error: 'Invalid agent slug' };
  if (!jobDescription || typeof jobDescription !== 'string' || !jobDescription.trim()) {
    return { error: 'Job description is required' };
  }

  // Verify user is assigned to this agent (or is hub admin)
  const isHubAdmin = process.env.SUPERADMIN_HUB === 'true' && user.role === 'admin';
  const assignedSlugs = new Set(user.assignedAgents ?? []);
  if (!isHubAdmin && !assignedSlugs.has(agentSlug)) {
    return { error: 'Not authorized for this agent' };
  }

  try {
    const { queryInstance } = await import('../superadmin/client.js');
    const { getInstanceRegistry } = await import('../superadmin/config.js');
    const registry = getInstanceRegistry();
    const instance = registry.find(i => i.name === agentSlug);
    if (!instance) return { error: `Agent "${agentSlug}" not found in registry` };

    // For local instance, use createJob directly
    if (instance.url === null) {
      const { createJob } = await import('../tools/create-job.js');
      const result = await createJob(jobDescription.trim(), {});
      return { job_id: result.job_id, branch: result.branch, agentSlug };
    }

    // For remote instance, POST to /api/jobs via spoke Bearer auth
    const url = new URL('/api/jobs', instance.url);
    const controller = new AbortController();
    const timeout = setTimeout(() => controller.abort(), 10000);
    try {
      const res = await fetch(url.toString(), {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
          Authorization: `Bearer ${instance.token}`,
        },
        body: JSON.stringify({ description: jobDescription.trim() }),
        signal: controller.signal,
      });
      if (!res.ok) {
        const body = await res.text().catch(() => '');
        return { error: `Agent returned ${res.status}: ${body}` };
      }
      const data = await res.json();
      return { job_id: data.job_id, branch: data.branch, agentSlug };
    } finally {
      clearTimeout(timeout);
    }
  } catch (err) {
    return { error: err.message || 'Failed to dispatch job' };
  }
}
```
