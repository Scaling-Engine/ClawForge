```javascript
  const db = getHubDb();

    createdAt: hubUsers.createdAt,
  }).from(hubUsers).where(eq(hubUsers.id, id)).get();

// 3. getAssignmentsForUser(userId) — all agent assignments for one user
export function getAssignmentsForUser(userId) {
  const db = getHubDb();
  return db.select()
    .from(agentAssignments)
    .where(eq(agentAssignments.userId, userId))
    .all();
}

// 4. upsertUserAssignment(userId, agentSlug, agentRole) — insert or update
// Uses select-then-update/insert (not INSERT OR REPLACE) to preserve createdAt
export function upsertUserAssignment(userId, agentSlug, agentRole) {
  const db = getHubDb();
  const existing = db.select()
    .from(agentAssignments)
    .where(
      and(
        eq(agentAssignments.userId, userId),
        eq(agentAssignments.agentSlug, agentSlug)
      )
    ).get();
  if (existing) {
    db.update(agentAssignments)
      .set({ agentRole })
      .where(eq(agentAssignments.id, existing.id))
      .run();
  } else {
    db.insert(agentAssignments).values({
      id: randomUUID(),
      userId,
      agentSlug,
      agentRole,
      createdAt: Date.now(),
    }).run();
  }
}

// 5. removeUserAssignment(userId, agentSlug) — delete a specific assignment
export function removeUserAssignment(userId, agentSlug) {
  const db = getHubDb();
  db.delete(agentAssignments)
    .where(
      and(
        eq(agentAssignments.userId, userId),
        eq(agentAssignments.agentSlug, agentSlug)
      )
    ).run();
}
```

```javascript
// ─────────────────────────────────────────────────────────────────────────────
// Hub user management — only available on hub instances (SUPERADMIN_HUB=true)

/**
 * Get all hub users (without password hashes). Admin-only.
 * @returns {Promise<Array<{id, email, role, createdAt}>>}
 */
export async function getHubUsers() {
  await requireAdmin();
  if (process.env.SUPERADMIN_HUB !== 'true') return [];
  const { getHubUsers: dbGetHubUsers } = await import('../db/hub-users.js');
  return dbGetHubUsers();
}

/**
 * Get agent assignments for a specific hub user. Admin-only.
 * @param {string} userId
 * @returns {Promise<Array<{id, userId, agentSlug, agentRole, createdAt}>>}
 */
export async function getUserAgentAssignments(userId) {
  await requireAdmin();
  if (process.env.SUPERADMIN_HUB !== 'true') return [];
  const { getAssignmentsForUser } = await import('../db/hub-users.js');
  return getAssignmentsForUser(userId);
}

/**
 * Get a single hub user by ID. Admin-only.
 * @param {string} userId
 * @returns {Promise<{id, email, role, createdAt}|null>}
 */
export async function getHubUserById(userId) {
  await requireAdmin();
  if (process.env.SUPERADMIN_HUB !== 'true') return null;
  const { getUserById } = await import('../db/hub-users.js');
  return getUserById(userId) ?? null;
}

/**
 * Save agent assignments for a user. Replaces all existing assignments.
 * assignments: Array<{agentSlug: string, agentRole: 'viewer'|'operator'|'admin'}>
 * Pass empty array to remove all assignments.
 * Admin-only. Takes effect immediately on the user's next request (JWT refresh).
 * @param {string} userId
 * @param {Array<{agentSlug: string, agentRole: string}>} assignments
 * @returns {Promise<{success: boolean} | {error: string}>}
 */
export async function setUserAgentAssignments(userId, assignments) {
  await requireAdmin();
  if (process.env.SUPERADMIN_HUB !== 'true') return { error: 'Not a hub instance' };
  const validRoles = ['viewer', 'operator', 'admin'];
  for (const a of assignments) {
    if (!a.agentSlug || !validRoles.includes(a.agentRole)) {
      return { error: `Invalid assignment: agentSlug=${a.agentSlug}, agentRole=${a.agentRole}` };
    }
  }
  const { getAssignmentsForUser, upsertUserAssignment, removeUserAssignment } = await import('../db/hub-users.js');
  const existing = getAssignmentsForUser(userId);
  const existingSlugs = new Set(existing.map(a => a.agentSlug));
  const newSlugs = new Set(assignments.map(a => a.agentSlug));
  // Remove assignments no longer present
  for (const slug of existingSlugs) {
    if (!newSlugs.has(slug)) removeUserAssignment(userId, slug);
  }
  // Upsert all new/updated assignments
  for (const { agentSlug, agentRole } of assignments) {
    upsertUserAssignment(userId, agentSlug, agentRole);
  }
  return { success: true };
}
```

```javascript
// ─────────────────────────────────────────────────────────────────────────────
// Agent picker data — health + stats for the authenticated user's assigned agents
// ─────────────────────────────────────────────────────────────────────────────

/**
 * Get picker-ready card data for all agents the current user is assigned to.
 * Filters queryAllInstances results to only the caller's assignedAgents.
 * Returns ALL instances for admins when SUPERADMIN_HUB=true (so superadmin sees all).
 * Each card: { name, status, lastJobAt, activeJobs, openPrs, activeWorkspaces, error }
 * @returns {Promise<Array<object>>}
 */
export async function getAgentPickerData() {
  const user = await requireAuth();
  const { queryAllInstances } = await import('../superadmin/client.js');
  const results = await queryAllInstances('health');

  // Filter to only instances this user is assigned to (admin sees all on hub)
  const isHubAdmin = process.env.SUPERADMIN_HUB === 'true' && user.role === 'admin';
  const assignedSlugs = new Set(user.assignedAgents ?? []);

  return results
    .filter(r => isHubAdmin || assignedSlugs.has(r.instance))
    .map(r => ({
      name: r.instance,
      status: r.data?.status ?? 'offline',
      lastJobAt: r.data?.lastJobAt ?? null,
      activeJobs: r.data?.activeJobs ?? 0,
      openPrs: r.data?.openPrs ?? 0,
      activeWorkspaces: r.data?.activeWorkspaces ?? 0,
      error: r.error ?? null,
    }));
}
```
