    **2. Add enforcement gate in createJobTool (BEFORE `const result = await createJob(enrichedDescription, ...)`  at ~line 79):**
    ```javascript
    // --- Billing enforcement gate (BILL-03, BILL-04) ---
    const instanceName = process.env.INSTANCE_NAME || 'noah';
    const limitCheck = checkUsageLimit(instanceName, 'jobs_per_month');

    // Hard limit rejection — return error before any GitHub API call
    if (!limitCheck.allowed) {
      return JSON.stringify({
        success: false,
        error: `Monthly job limit reached for this instance. ` +
               `Current usage: ${limitCheck.current} jobs. Limit: ${limitCheck.limit} jobs. ` +
               `Resets (UTC): ${limitCheck.resetDate}`,
      });
    }

    // 80% warning — send Slack to operator channel, but allow the job
    const periodMonth = new Date().toISOString().slice(0, 7);
    if (limitCheck.percentUsed >= 0.8 && limitCheck.limit !== null && !wasWarningSent(instanceName, periodMonth)) {
      try {
        const { SLACK_BOT_TOKEN, SLACK_OPERATOR_CHANNEL } = process.env;
        if (SLACK_BOT_TOKEN && SLACK_OPERATOR_CHANNEL) {
          const { WebClient } = await import('@slack/web-api');
          const slack = new WebClient(SLACK_BOT_TOKEN);
          await slack.chat.postMessage({
            channel: SLACK_OPERATOR_CHANNEL,
            text: `Warning: Instance "${instanceName}" has used ${limitCheck.current}/${limitCheck.limit} jobs this month (${Math.round(limitCheck.percentUsed * 100)}%). Limit resets (UTC): ${limitCheck.resetDate}`,
          });
        }
        markWarningSent(instanceName, periodMonth);
      } catch (err) {
        console.warn('Failed to send billing warning:', err.message);
        // Non-fatal — allow the job to proceed
      }
    }
    ```

    Note: The `SLACK_OPERATOR_CHANNEL` env var is used to send warnings to a known operator channel, not the user's thread. If the env var is not set, the warning is silently skipped. This is a new env var — document in the summary.
