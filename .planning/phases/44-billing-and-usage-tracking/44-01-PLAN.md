    1. Add two new table definitions to `lib/db/schema.js` (append after `errorLog`):

    ```javascript
    export const usageEvents = sqliteTable('usage_events', {
      id: text('id').primaryKey(),
      instanceName: text('instance_name').notNull(),
      eventType: text('event_type').notNull(),       // 'job_dispatch'
      quantity: real('quantity').notNull().default(1),
      durationSeconds: integer('duration_seconds'),   // nullable — populated on job completion
      periodMonth: text('period_month').notNull(),     // 'YYYY-MM'
      refId: text('ref_id'),                          // jobId for tracing
      createdAt: integer('created_at').notNull(),
    });

    export const billingLimits = sqliteTable('billing_limits', {
      id: text('id').primaryKey(),
      instanceName: text('instance_name').notNull(),
      limitType: text('limit_type').notNull(),         // 'jobs_per_month', 'concurrent_jobs'
      limitValue: real('limit_value').notNull(),
      warningSentPeriod: text('warning_sent_period'),  // 'YYYY-MM' — dedup 80% warnings
      createdAt: integer('created_at').notNull(),
      updatedAt: integer('updated_at').notNull(),
    });
    ```

    2. Create `lib/db/usage.js` with these exports, following the exact pattern from `lib/db/error-log.js`:
    - `recordUsageEvent({ instanceName, eventType, quantity, durationSeconds, refId, periodMonth })` — inserts row with randomUUID id and Date.now() createdAt
    - `getUsageSummary(instanceName, periodMonth)` — returns `{ jobCount, totalDurationSeconds }` using COUNT(*) and SUM(duration_seconds) with WHERE on instanceName + eventType='job_dispatch' + periodMonth
    - `getBillingLimits(instanceName)` — returns `{ jobsPerMonth: number|null, concurrentJobs: number|null }` by querying billingLimits WHERE instanceName, mapping limitType to keys
    - `upsertBillingLimit(instanceName, limitType, limitValue)` — checks for existing row, updates if found, inserts if not
    - `markWarningSent(instanceName, periodMonth)` — updates warningSentPeriod on the 'jobs_per_month' row
    - `wasWarningSent(instanceName, periodMonth)` — returns boolean checking warningSentPeriod === periodMonth

    All functions are synchronous (better-sqlite3 via drizzle `.run()` / `.get()` / `.all()`). Use `import { eq, and, sql } from 'drizzle-orm'` for queries.

    3. Create `lib/billing/enforce.js` with one export:
    - `checkUsageLimit(instanceName, limitType = 'jobs_per_month')` — returns `{ allowed, current, limit, remaining, percentUsed, resetDate }`
    - When no limit row exists: returns `{ allowed: true, current: N, limit: null, remaining: null, percentUsed: 0, resetDate: null }`
    - When limit exists: `allowed = current < limit`, `percentUsed = current / limit`, `resetDate = first day of next month UTC in 'YYYY-MM-DD' format`
    - Reset date calculation: `new Date(Date.UTC(now.getUTCFullYear(), now.getUTCMonth() + 1, 1)).toISOString().slice(0, 10)`
