1. In lib/chat/actions.js, add a new server action `getJobAuthMethod()` after the `getConfigValues` function (around line 1068):

```javascript
/**
 * Detect which auth method job containers will use.
 * Reads AGENT_LLM_SECRETS env var to check for token presence.
 * @returns {Promise<{method: string, detail: string}>}
 */
export async function getJobAuthMethod() {
  await requireAdmin();
  try {
    const secrets = JSON.parse(process.env.AGENT_LLM_SECRETS || '{}');
    const hasOAuth = !!secrets.CLAUDE_CODE_OAUTH_TOKEN;
    const hasApiKey = !!secrets.ANTHROPIC_API_KEY;
    if (hasOAuth && hasApiKey) {
      return { method: 'subscription', detail: 'CLAUDE_CODE_OAUTH_TOKEN set (ANTHROPIC_API_KEY also present as fallback)' };
    }
    if (hasOAuth) {
      return { method: 'subscription', detail: 'CLAUDE_CODE_OAUTH_TOKEN' };
    }
    if (hasApiKey) {
      return { method: 'api-key', detail: 'ANTHROPIC_API_KEY' };
    }
    return { method: 'none', detail: 'No auth configured in AGENT_LLM_SECRETS' };
  } catch {
    return { method: 'unknown', detail: 'Could not parse AGENT_LLM_SECRETS' };
  }
}
```
